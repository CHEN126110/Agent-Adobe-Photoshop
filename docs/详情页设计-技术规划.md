## 一、业务需求

### 详情页结构定义

```
详情页设计结构:
├── A 营销信息 (促销/活动)
├── B 信任状 / 品牌背书
├── C 详情页首屏（核心卖点）
├── D 图标 Icon（卖点符号化）
├── E KV 图（氛围/视觉锤）
├── F 颜色款式展示
├── G 面料 / 材质说明
├── H 解决痛点问题
├── I 穿搭推荐
├── J 细节展示
├── K 产品信息（面料成分等）
├── 模特场景实拍图
├── 售后服务
└── ... (AI 可延展)
```

### 模板图层结构

```
图层组 [Screen-XX 屏名称]    ← 每屏 = 一个图层组
├── 图层组 [文案]
│   ├── 文本图层 [主标题]
│   └── 文本图层 [副标题]
├── 图层组 [辅助图标]
│   └── 图层 [icon]
└── 图层组 [图片]
    ├── 图层 [产品图]         ← 剪切蒙版 (clipped)
    └── 矩形图层 [显示范围]    ← 剪切基底 (base)
```

### 4.1 模板解析器 (DetailPageParser)

**文件**: `DesignEcho-UXP/src/tools/layout/detail-page-parser.ts`

**功能**:
- 遍历顶层图层组，识别每屏结构
- 根据命名规则推断屏类型
- 识别文案/图片/图标占位符
- 检测剪切蒙版关系

**识别策略**:

| 目标 | 策略 |
|------|------|
| 屏类型 | 名称正则匹配 + 关键词 |
| 文案组 | 包含 `text` 类型图层的组 |
| 图片组 | 包含剪切蒙版关系的组 |
| 图标组 | 包含小尺寸形状图层的组 |

**核心实现**:

```typescript
// detail-page-parser.ts
import { app } from 'photoshop';
import type { 
    ParsedScreen, ScreenType, TemplateParseResult,
    CopyPlaceholder, ImagePlaceholder, IconPlaceholder,
    ScreenTypeConfig, BoundingBox
} from './detail-page.types';

// 屏类型配置表
const SCREEN_TYPE_CONFIGS: ScreenTypeConfig[] = [
    {
        type: 'A_营销信息',
        namePatterns: [/营销/i, /活动/i, /优惠/i, /促销/i],
        keywords: ['满减', '折扣', '限时', '新品'],
        recommendedAssetType: 'scene',
        knowledgeTypes: ['marketing', 'promotion']
    },
    {
        type: 'C_核心卖点',
        namePatterns: [/首屏/i, /核心/i, /卖点/i, /主图/i],
        keywords: ['核心', '卖点', '特点', '优势'],
        recommendedAssetType: 'product',
        knowledgeTypes: ['selling_point', 'feature']
    },
    {
        type: 'G_面料说明',
        namePatterns: [/面料/i, /材质/i, /成分/i],
        keywords: ['棉', '涤纶', '透气', '舒适'],
        recommendedAssetType: 'detail',
        knowledgeTypes: ['material', 'fabric']
    },
    {
        type: 'H_解决痛点',
        namePatterns: [/痛点/i, /问题/i, /解决/i],
        keywords: ['不会', '不怕', '解决', '告别'],
        recommendedAssetType: 'detail',
        knowledgeTypes: ['pain_point', 'solution']
    },
    // ... 其他类型配置
];

export class DetailPageParser {
    
    /**
     * 解析当前文档的详情页模板结构
     */
    async parse(): Promise<TemplateParseResult> {
        const startTime = Date.now();
        const doc = app.activeDocument;
        
        if (!doc) {
            return { success: false, documentName: '', documentSize: { width: 0, height: 0 }, 
                     screenCount: 0, screens: [], issues: [], parseTime: 0 };
        }
        
        const screens: ParsedScreen[] = [];
        const issues: LayerIssue[] = [];
        let screenIndex = 0;
        
        // 遍历顶层图层组
        for (const layer of doc.layers) {
            if (layer.kind === 'group') {
                const screen = await this.parseScreen(layer, screenIndex);
                screens.push(screen);
                screenIndex++;
            }
        }
        
        return {
            success: true,
            documentName: doc.name,
            documentSize: { width: doc.width, height: doc.height },
            screenCount: screens.length,
            screens,
            issues,
            parseTime: Date.now() - startTime
        };
    }
    
    /**
     * 解析单个屏 (图层组)
     */
    private async parseScreen(group: Layer, index: number): Promise<ParsedScreen> {
        const bounds = this.getBounds(group);
        const { type, confidence } = this.inferScreenType(group.name);
        
        const copyPlaceholders: CopyPlaceholder[] = [];
        const imagePlaceholders: ImagePlaceholder[] = [];
        const iconPlaceholders: IconPlaceholder[] = [];
        
        // 递归遍历子图层
        await this.traverseLayers(group.layers, {
            onTextLayer: (layer) => {
                copyPlaceholders.push(this.parseTextLayer(layer));
            },
            onClippingBase: (baseLayer, clippedLayer) => {
                imagePlaceholders.push(this.parseImagePlaceholder(baseLayer, clippedLayer));
            },
            onSmallShape: (layer) => {
                iconPlaceholders.push(this.parseIconPlaceholder(layer));
            }
        });
        
        return {
            id: group.id,
            name: group.name,
            type,
            typeConfidence: confidence,
            index,
            bounds,
            visible: group.visible,
            copyPlaceholders,
            imagePlaceholders,
            iconPlaceholders
        };
    }
    
    /**
     * 推断屏类型
     */
    private inferScreenType(name: string): { type: ScreenType; confidence: number } {
        for (const config of SCREEN_TYPE_CONFIGS) {
            // 正则匹配
            for (const pattern of config.namePatterns) {
                if (pattern.test(name)) {
                    return { type: config.type, confidence: 0.9 };
                }
            }
            // 关键词匹配
            for (const keyword of config.keywords) {
                if (name.includes(keyword)) {
                    return { type: config.type, confidence: 0.7 };
                }
            }
        }
        return { type: 'CUSTOM', confidence: 0.3 };
    }
    
    /**
     * 解析文案图层
     */
    private parseTextLayer(layer: Layer): CopyPlaceholder {
        const bounds = this.getBounds(layer);
        const fontSize = layer.textItem?.characterStyle?.size || 16;
        
        // 根据字号推断角色
        let role: CopyPlaceholder['role'] = 'body';
        if (fontSize >= 36) role = 'title';
        else if (fontSize >= 24) role = 'subtitle';
        else if (fontSize <= 12) role = 'label';
        
        return {
            layerId: layer.id,
            layerName: layer.name,
            currentText: layer.textItem?.contents || '',
            bounds,
            role,
            fontSize,
            fontFamily: layer.textItem?.characterStyle?.fontName,
            maxWidth: bounds.width
        };
    }
    
    /**
     * 解析图片占位符 (剪切蒙版)
     */
    private parseImagePlaceholder(baseLayer: Layer, clippedLayer?: Layer): ImagePlaceholder {
        const bounds = this.getBounds(baseLayer);
        const aspectRatio = bounds.width / bounds.height;
        
        // 根据宽高比推荐素材类型
        let recommendedAssetType: AssetType = 'product';
        if (aspectRatio > 1.5) recommendedAssetType = 'scene';
        else if (aspectRatio < 0.7) recommendedAssetType = 'model';
        
        return {
            layerId: clippedLayer?.id || baseLayer.id,
            layerName: clippedLayer?.name || baseLayer.name,
            baseLayerId: baseLayer.id,
            baseLayerName: baseLayer.name,
            bounds,
            isClippingMask: !!clippedLayer,
            clippingInfo: clippedLayer ? {
                isClipped: true,
                baseLayerId: baseLayer.id,
                baseBounds: bounds
            } : undefined,
            recommendedAssetType,
            aspectRatio
        };
    }
    
    private getBounds(layer: Layer): BoundingBox {
        const b = layer.bounds;
        return {
            left: b.left,
            top: b.top,
            right: b.right,
            bottom: b.bottom,
            width: b.right - b.left,
            height: b.bottom - b.top
        };
    }
}
```

### 4.2 图层关系检测器 (LayerRelationDetector)

**文件**: `DesignEcho-UXP/src/tools/layout/layer-relation-detector.ts`

**检测项**:

| 异常类型 | 检测方法 | 严重程度 |
|---------|---------|---------|
| 图层遮挡 | bounds 重叠 + 透明度检查 | warning |
| 剪切蒙版断裂 | clipped 属性 + 顺序检查 | critical |
| 文字溢出 | bounds vs 容器边界 | warning |
| 图片变形 | 宽高比检测 | warning |
| 效果截断 | 效果范围 vs 画布 | info |

**核心实现**:

```typescript
// layer-relation-detector.ts
import { app, action } from 'photoshop';
import type { LayerIssue, LayerIssueType, ParsedScreen, BoundingBox } from './detail-page.types';

export class LayerRelationDetector {
    
    /**
     * 检测指定屏的所有图层问题
     */
    async detectIssues(screen: ParsedScreen): Promise<LayerIssue[]> {
        const issues: LayerIssue[] = [];
        const doc = app.activeDocument;
        const screenGroup = doc.layers.find(l => l.id === screen.id);
        
        if (!screenGroup) return issues;
        
        // 1. 检测剪切蒙版问题
        issues.push(...await this.detectClippingIssues(screenGroup));
        
        // 2. 检测文字溢出
        for (const copy of screen.copyPlaceholders) {
            const overflow = await this.detectTextOverflow(copy);
            if (overflow) issues.push(overflow);
        }
        
        // 3. 检测图层遮挡
        issues.push(...await this.detectOcclusion(screenGroup));
        
        // 4. 检测图片变形
        for (const img of screen.imagePlaceholders) {
            const distortion = await this.detectAspectDistortion(img);
            if (distortion) issues.push(distortion);
        }
        
        return issues;
    }
    
    /**
     * 检测剪切蒙版断裂
     */
    private async detectClippingIssues(group: Layer): Promise<LayerIssue[]> {
        const issues: LayerIssue[] = [];
        
        const layers = group.layers || [];
        for (let i = 0; i < layers.length; i++) {
            const layer = layers[i];
            
            // 检查是否应该有剪切蒙版但没有
            if (this.shouldBeClipped(layer) && !layer.clipped) {
                issues.push({
                    type: 'clipping_broken',
                    severity: 'critical',
                    layerId: layer.id,
                    layerName: layer.name,
                    description: `图层 "${layer.name}" 剪切蒙版关系断裂`,
                    autoFixable: true,
                    suggestedFix: '重新建立剪切蒙版',
                    fixParams: { 
                        baseLayerIndex: i + 1  // 下一个图层应为基底
                    }
                });
            }
            
            // 递归检查子组
            if (layer.kind === 'group') {
                issues.push(...await this.detectClippingIssues(layer));
            }
        }
        
        return issues;
    }
    
    /**
     * 检测文字溢出
     */
    private async detectTextOverflow(copy: CopyPlaceholder): Promise<LayerIssue | null> {
        const doc = app.activeDocument;
        const layer = this.findLayerById(doc.layers, copy.layerId);
        
        if (!layer || layer.kind !== 'text') return null;
        
        const bounds = layer.bounds;
        const actualWidth = bounds.right - bounds.left;
        
        // 如果实际宽度超过容器宽度 10% 以上
        if (copy.maxWidth && actualWidth > copy.maxWidth * 1.1) {
            return {
                type: 'overflow',
                severity: 'warning',
                layerId: copy.layerId,
                layerName: copy.layerName,
                description: `文字 "${copy.layerName}" 超出容器边界`,
                autoFixable: true,
                suggestedFix: '缩小字号或调整文本',
                fixParams: {
                    currentWidth: actualWidth,
                    maxWidth: copy.maxWidth,
                    overflowRatio: actualWidth / copy.maxWidth
                }
            };
        }
        
        return null;
    }
    
    /**
     * 检测图层遮挡
     */
    private async detectOcclusion(group: Layer): Promise<LayerIssue[]> {
        const issues: LayerIssue[] = [];
        const layers = group.layers || [];
        
        for (let i = 0; i < layers.length; i++) {
            for (let j = i + 1; j < layers.length; j++) {
                const upperLayer = layers[i];  // 上层 (先绘制)
                const lowerLayer = layers[j];  // 下层
                
                // 跳过隐藏图层
                if (!upperLayer.visible || !lowerLayer.visible) continue;
                
                // 检查是否重叠
                if (this.boundsOverlap(upperLayer.bounds, lowerLayer.bounds)) {
                    // 如果上层是不透明的且完全覆盖下层
                    if (upperLayer.opacity === 100 && 
                        this.fullyCovers(upperLayer.bounds, lowerLayer.bounds)) {
                        issues.push({
                            type: 'occlusion',
                            severity: 'warning',
                            layerId: lowerLayer.id,
                            layerName: lowerLayer.name,
                            description: `图层 "${lowerLayer.name}" 被 "${upperLayer.name}" 完全遮挡`,
                            autoFixable: true,
                            suggestedFix: '调整图层顺序或减少上层透明度',
                            fixParams: {
                                occludingLayerId: upperLayer.id,
                                occludingLayerName: upperLayer.name
                            }
                        });
                    }
                }
            }
        }
        
        return issues;
    }
    
    /**
     * 检测图片宽高比变形
     */
    private async detectAspectDistortion(img: ImagePlaceholder): Promise<LayerIssue | null> {
        const doc = app.activeDocument;
        const layer = this.findLayerById(doc.layers, img.layerId);
        
        if (!layer) return null;
        
        // 获取原始宽高比 (通过 batchPlay)
        const originalRatio = await this.getOriginalAspectRatio(layer.id);
        const currentBounds = layer.bounds;
        const currentRatio = (currentBounds.right - currentBounds.left) / 
                            (currentBounds.bottom - currentBounds.top);
        
        // 允许 5% 误差
        if (originalRatio && Math.abs(currentRatio - originalRatio) / originalRatio > 0.05) {
            return {
                type: 'aspect_distortion',
                severity: 'warning',
                layerId: img.layerId,
                layerName: img.layerName,
                description: `图片 "${img.layerName}" 宽高比变形 (${currentRatio.toFixed(2)} → ${originalRatio.toFixed(2)})`,
                autoFixable: true,
                suggestedFix: '恢复原始宽高比',
                fixParams: {
                    originalRatio,
                    currentRatio
                }
            };
        }
        
        return null;
    }
    
    private boundsOverlap(a: any, b: any): boolean {
        return !(a.right < b.left || a.left > b.right || 
                 a.bottom < b.top || a.top > b.bottom);
    }
    
    private fullyCovers(outer: any, inner: any): boolean {
        return outer.left <= inner.left && outer.right >= inner.right &&
               outer.top <= inner.top && outer.bottom >= inner.bottom;
    }
    
    private findLayerById(layers: any[], id: number): Layer | null {
        for (const layer of layers) {
            if (layer.id === id) return layer;
            if (layer.layers) {
                const found = this.findLayerById(layer.layers, id);
                if (found) return found;
            }
        }
        return null;
    }
    
    private shouldBeClipped(layer: Layer): boolean {
        // 根据命名规则判断是否应该被裁剪
        const name = layer.name.toLowerCase();
        return name.includes('图片') || name.includes('image') || 
               name.includes('photo') || name.includes('img');
    }
    
    private async getOriginalAspectRatio(layerId: number): Promise<number | null> {
        try {
            const result = await action.batchPlay([{
                _obj: 'get',
                _target: [{ _ref: 'layer', _id: layerId }],
                _options: { dialogOptions: 'dontDisplay' }
            }], { synchronousExecution: true });
            
            const smartObject = result[0]?.smartObject;
            if (smartObject) {
                return smartObject.width / smartObject.height;
            }
            return null;
        } catch {
            return null;
        }
    }
}
```

### 4.3 自动修复引擎 (AutoFixer)

**文件**: `DesignEcho-UXP/src/tools/layout/auto-fixer.ts`

**修复能力**:

| 问题类型 | 修复方法 |
|---------|---------|
| 剪切蒙版断裂 | 释放→调整顺序→重建 |
| 文字溢出 | 缩小字号 (保持可读) |
| 图片变形 | 恢复原始宽高比 |
| 图层遮挡 | 调整顺序/透明度 |

**核心实现**:

```typescript
// auto-fixer.ts
import { app, action, core } from 'photoshop';
import type { LayerIssue, FixResult, LayerIssueType } from './detail-page.types';

export class AutoFixer {
    
    private readonly MIN_FONT_SIZE = 10;  // 最小可读字号
    
    /**
     * 修复所有可自动修复的问题
     */
    async fixAll(issues: LayerIssue[]): Promise<FixResult[]> {
        const results: FixResult[] = [];
        
        // 按优先级排序: critical > warning > info
        const sorted = issues
            .filter(i => i.autoFixable)
            .sort((a, b) => this.severityOrder(a.severity) - this.severityOrder(b.severity));
        
        for (const issue of sorted) {
            const result = await this.fixIssue(issue);
            results.push(result);
        }
        
        return results;
    }
    
    /**
     * 修复单个问题
     */
    async fixIssue(issue: LayerIssue): Promise<FixResult> {
        try {
            switch (issue.type) {
                case 'clipping_broken':
                    return await this.fixClippingMask(issue);
                case 'overflow':
                    return await this.fixTextOverflow(issue);
                case 'aspect_distortion':
                    return await this.fixAspectRatio(issue);
                case 'occlusion':
                    return await this.fixOcclusion(issue);
                default:
                    return {
                        issueType: issue.type,
                        layerId: issue.layerId,
                        success: false,
                        message: `不支持自动修复类型: ${issue.type}`
                    };
            }
        } catch (e: any) {
            return {
                issueType: issue.type,
                layerId: issue.layerId,
                success: false,
                message: `修复失败: ${e.message}`
            };
        }
    }
    
    /**
     * 修复剪切蒙版断裂
     */
    private async fixClippingMask(issue: LayerIssue): Promise<FixResult> {
        const changes: string[] = [];
        
        await core.executeAsModal(async () => {
            const doc = app.activeDocument;
            const layer = this.findLayerById(doc.layers, issue.layerId);
            
            if (!layer) throw new Error('图层不存在');
            
            // 1. 如果已有剪切蒙版，先释放
            if (layer.clipped) {
                await action.batchPlay([{
                    _obj: 'ungroup',
                    _target: [{ _ref: 'layer', _id: layer.id }]
                }], { synchronousExecution: true });
                changes.push('释放现有剪切蒙版');
            }
            
            // 2. 确定基底图层
            const baseLayerIndex = issue.fixParams?.baseLayerIndex;
            const parent = layer.parent;
            if (!parent || !parent.layers) throw new Error('无法确定父级');
            
            const currentIndex = parent.layers.findIndex((l: any) => l.id === layer.id);
            const baseLayer = parent.layers[currentIndex + 1];
            
            if (!baseLayer) throw new Error('找不到基底图层');
            
            // 3. 重建剪切蒙版
            await action.batchPlay([{
                _obj: 'groupEvent',
                _target: [{ _ref: 'layer', _id: layer.id }]
            }], { synchronousExecution: true });
            changes.push(`建立剪切蒙版到 "${baseLayer.name}"`);
            
        }, { commandName: '修复剪切蒙版' });
        
        return {
            issueType: 'clipping_broken',
            layerId: issue.layerId,
            success: true,
            message: '剪切蒙版已修复',
            changes
        };
    }
    
    /**
     * 修复文字溢出
     */
    private async fixTextOverflow(issue: LayerIssue): Promise<FixResult> {
        const changes: string[] = [];
        const params = issue.fixParams || {};
        
        await core.executeAsModal(async () => {
            const doc = app.activeDocument;
            const layer = this.findLayerById(doc.layers, issue.layerId);
            
            if (!layer || layer.kind !== 'text') throw new Error('非文本图层');
            
            const textItem = layer.textItem;
            const currentSize = textItem.characterStyle.size;
            const overflowRatio = params.overflowRatio || 1.2;
            
            // 计算需要缩小的比例
            const targetSize = Math.max(
                this.MIN_FONT_SIZE,
                Math.floor(currentSize / overflowRatio)
            );
            
            // 应用新字号
            textItem.characterStyle.size = targetSize;
            changes.push(`字号从 ${currentSize}px 缩小到 ${targetSize}px`);
            
        }, { commandName: '修复文字溢出' });
        
        return {
            issueType: 'overflow',
            layerId: issue.layerId,
            success: true,
            message: '文字溢出已修复',
            changes
        };
    }
    
    /**
     * 修复宽高比变形
     */
    private async fixAspectRatio(issue: LayerIssue): Promise<FixResult> {
        const changes: string[] = [];
        const params = issue.fixParams || {};
        
        await core.executeAsModal(async () => {
            const doc = app.activeDocument;
            const layer = this.findLayerById(doc.layers, issue.layerId);
            
            if (!layer) throw new Error('图层不存在');
            
            const originalRatio = params.originalRatio;
            const currentRatio = params.currentRatio;
            
            if (!originalRatio) throw new Error('缺少原始宽高比');
            
            const bounds = layer.bounds;
            const currentWidth = bounds.right - bounds.left;
            const currentHeight = bounds.bottom - bounds.top;
            
            // 计算修正值 (保持宽度不变，调整高度)
            const targetHeight = currentWidth / originalRatio;
            const scaleY = (targetHeight / currentHeight) * 100;
            
            // 使用 transform 命令
            await action.batchPlay([{
                _obj: 'transform',
                _target: [{ _ref: 'layer', _id: layer.id }],
                freeTransformCenterState: { _enum: 'quadCenterState', _value: 'QCSAverage' },
                width: { _unit: 'percentUnit', _value: 100 },
                height: { _unit: 'percentUnit', _value: scaleY },
                interfaceIconFrameDimmed: { _enum: 'interpolationType', _value: 'bicubic' }
            }], { synchronousExecution: true });
            
            changes.push(`宽高比从 ${currentRatio.toFixed(2)} 恢复到 ${originalRatio.toFixed(2)}`);
            
        }, { commandName: '修复宽高比' });
        
        return {
            issueType: 'aspect_distortion',
            layerId: issue.layerId,
            success: true,
            message: '宽高比已修复',
            changes
        };
    }
    
    /**
     * 修复图层遮挡
     */
    private async fixOcclusion(issue: LayerIssue): Promise<FixResult> {
        const changes: string[] = [];
        const params = issue.fixParams || {};
        
        await core.executeAsModal(async () => {
            const doc = app.activeDocument;
            const occludingLayerId = params.occludingLayerId;
            
            if (!occludingLayerId) throw new Error('缺少遮挡图层信息');
            
            const occludingLayer = this.findLayerById(doc.layers, occludingLayerId);
            
            if (!occludingLayer) throw new Error('遮挡图层不存在');
            
            // 策略1: 降低遮挡图层透明度
            if (occludingLayer.opacity === 100) {
                occludingLayer.opacity = 80;
                changes.push(`降低 "${occludingLayer.name}" 透明度到 80%`);
            }
            
        }, { commandName: '修复图层遮挡' });
        
        return {
            issueType: 'occlusion',
            layerId: issue.layerId,
            success: true,
            message: '图层遮挡已处理',
            changes
        };
    }
    
    private severityOrder(severity: string): number {
        switch (severity) {
            case 'critical': return 0;
            case 'warning': return 1;
            case 'info': return 2;
            default: return 3;
        }
    }
    
    private findLayerById(layers: any[], id: number): any {
        for (const layer of layers) {
            if (layer.id === id) return layer;
            if (layer.layers) {
                const found = this.findLayerById(layer.layers, id);
                if (found) return found;
            }
        }
        return null;
    }
}
```

**核心实现**:

```typescript
// content-matcher.ts (Agent 侧)
import { RAGService } from '../rag/rag-service';
import { EcommerceProjectService } from '../ecommerce-project-service';
import type { 
    ParsedScreen, FillPlan, CopyFillItem, ImageFillItem,
    ScreenType, AssetType
} from '@shared/types/detail-page.types';

// 屏类型到知识库类型的映射
const SCREEN_TO_KNOWLEDGE_MAP: Record<ScreenType, string[]> = {
    'A_营销信息': ['promotion', 'discount', 'event'],
    'B_信任状': ['brand', 'certification', 'award'],
    'C_核心卖点': ['selling_point', 'feature', 'benefit'],
    'D_图标卖点': ['icon', 'quick_point'],
    'E_KV图': ['kv', 'hero'],
    'F_颜色展示': ['color', 'variant'],
    'G_面料说明': ['material', 'fabric', 'composition'],
    'H_解决痛点': ['pain_point', 'solution', 'problem'],
    'I_穿搭推荐': ['styling', 'outfit', 'match'],
    'J_细节展示': ['detail', 'craftsmanship', 'closeup'],
    'K_产品信息': ['specification', 'size', 'info'],
    'L_模特实拍': ['model', 'lifestyle'],
    'M_售后服务': ['service', 'guarantee', 'policy'],
    'CUSTOM': []
};

// 屏类型到素材类型的映射
const SCREEN_TO_ASSET_MAP: Record<ScreenType, AssetType[]> = {
    'A_营销信息': ['scene'],
    'B_信任状': ['icon'],
    'C_核心卖点': ['product'],
    'D_图标卖点': ['icon'],
    'E_KV图': ['scene', 'product'],
    'F_颜色展示': ['product'],
    'G_面料说明': ['detail'],
    'H_解决痛点': ['detail', 'product'],
    'I_穿搭推荐': ['model', 'scene'],
    'J_细节展示': ['detail'],
    'K_产品信息': ['product'],
    'L_模特实拍': ['model'],
    'M_售后服务': ['icon'],
    'CUSTOM': ['product']
};

export class ContentMatcher {
    
    constructor(
        private ragService: RAGService,
        private projectService: EcommerceProjectService
    ) {}
    
    /**
     * 为所有屏生成填充方案
     */
    async generateFillPlans(
        screens: ParsedScreen[], 
        projectPath: string
    ): Promise<FillPlan[]> {
        const plans: FillPlan[] = [];
        
        // 扫描项目素材
        const projectAssets = await this.projectService.scanProject(projectPath);
        
        for (const screen of screens) {
            const plan = await this.generateScreenPlan(screen, projectAssets);
            plans.push(plan);
        }
        
        return plans;
    }
    
    /**
     * 为单个屏生成填充方案
     */
    private async generateScreenPlan(
        screen: ParsedScreen,
        projectAssets: any
    ): Promise<FillPlan> {
        const copies: CopyFillItem[] = [];
        const images: ImageFillItem[] = [];
        
        // 1. 匹配文案
        for (const copy of screen.copyPlaceholders) {
            const matched = await this.matchCopy(copy, screen.type);
            copies.push(matched);
        }
        
        // 2. 匹配图片
        for (const img of screen.imagePlaceholders) {
            const matched = await this.matchImage(img, screen.type, projectAssets);
            images.push(matched);
        }
        
        // 计算置信度
        const avgConfidence = this.calculateConfidence(copies, images);
        
        return {
            screenId: screen.id,
            screenName: screen.name,
            screenType: screen.type,
            copies,
            images,
            confidence: avgConfidence,
            needsReview: avgConfidence < 0.7
        };
    }
    
    /**
     * 匹配文案内容
     */
    private async matchCopy(
        placeholder: CopyPlaceholder, 
        screenType: ScreenType
    ): Promise<CopyFillItem> {
        const knowledgeTypes = SCREEN_TO_KNOWLEDGE_MAP[screenType] || [];
        
        // 从知识库检索
        const query = `${screenType} ${placeholder.role} ${placeholder.currentText}`;
        const results = await this.ragService.search(query, {
            filter: { types: knowledgeTypes },
            limit: 3
        });
        
        if (results.length > 0 && results[0].score > 0.7) {
            return {
                layerId: placeholder.layerId,
                layerName: placeholder.layerName,
                content: results[0].text,
                source: 'knowledge',
                sourceId: results[0].id,
                originalText: placeholder.currentText
            };
        }
        
        // 知识库无匹配时，保留原文案
        return {
            layerId: placeholder.layerId,
            layerName: placeholder.layerName,
            content: placeholder.currentText,
            source: 'template',
            originalText: placeholder.currentText
        };
    }
    
    /**
     * 匹配图片素材
     */
    private async matchImage(
        placeholder: ImagePlaceholder,
        screenType: ScreenType,
        projectAssets: any
    ): Promise<ImageFillItem> {
        const preferredTypes = SCREEN_TO_ASSET_MAP[screenType] || ['product'];
        
        // 按优先级查找素材
        for (const assetType of preferredTypes) {
            const candidates = projectAssets.images?.filter(
                (img: any) => img.type === assetType
            ) || [];
            
            if (candidates.length > 0) {
                // 选择宽高比最接近的
                const best = this.findBestAspectRatioMatch(
                    candidates, 
                    placeholder.aspectRatio
                );
                
                return {
                    layerId: placeholder.layerId,
                    layerName: placeholder.layerName,
                    imagePath: best.path,
                    fillMode: 'smart',
                    assetType,
                    needsMatting: assetType === 'product',
                    subjectAlign: 'center'
                };
            }
        }
        
        // 无匹配时返回空填充
        return {
            layerId: placeholder.layerId,
            layerName: placeholder.layerName,
            imagePath: '',
            fillMode: 'cover',
            assetType: 'product'
        };
    }
    
    /**
     * 找到宽高比最接近的图片
     */
    private findBestAspectRatioMatch(candidates: any[], targetRatio: number): any {
        let best = candidates[0];
        let minDiff = Infinity;
        
        for (const img of candidates) {
            const ratio = img.width / img.height;
            const diff = Math.abs(ratio - targetRatio);
            if (diff < minDiff) {
                minDiff = diff;
                best = img;
            }
        }
        
        return best;
    }
    
    /**
     * 计算匹配置信度
     */
    private calculateConfidence(copies: CopyFillItem[], images: ImageFillItem[]): number {
        let total = 0;
        let matched = 0;
        
        for (const copy of copies) {
            total++;
            if (copy.source === 'knowledge') matched++;
        }
        
        for (const img of images) {
            total++;
            if (img.imagePath) matched++;
        }
        
        return total > 0 ? matched / total : 0;
    }
}
```

### 4.5 批量填充工具 (DetailPageFiller)

**文件**: `DesignEcho-UXP/src/tools/layout/detail-page-filler.ts`

**执行步骤**:
1. 遍历填充方案
2. 替换文案 (setTextContent)
3. 填充图片 (placeImage + createClippingMask)
4. 调整位置/缩放

**核心实现**:

```typescript
// detail-page-filler.ts (UXP 侧)
import { app, action, core } from 'photoshop';
import type { FillPlan, CopyFillItem, ImageFillItem } from './detail-page.types';

const uxp = require('uxp');
const fs = uxp.storage.localFileSystem;

export class DetailPageFiller {
    
    /**
     * 执行填充方案
     */
    async fill(plan: FillPlan): Promise<{ success: boolean; errors: string[] }> {
        const errors: string[] = [];
        const doc = app.activeDocument;
        
        // 1. 填充文案
        for (const copy of plan.copies) {
            try {
                await this.fillCopy(copy);
            } catch (e: any) {
                errors.push(`文案填充失败 [${copy.layerName}]: ${e.message}`);
            }
        }
        
        // 2. 填充图片
        for (const image of plan.images) {
            if (!image.imagePath) continue;  // 跳过空路径
            
            try {
                await this.fillImage(image);
            } catch (e: any) {
                errors.push(`图片填充失败 [${image.layerName}]: ${e.message}`);
            }
        }
        
        return {
            success: errors.length === 0,
            errors
        };
    }
    
    /**
     * 填充文案
     */
    private async fillCopy(item: CopyFillItem): Promise<void> {
        await core.executeAsModal(async () => {
            const doc = app.activeDocument;
            const layer = this.findLayerById(doc.layers, item.layerId);
            
            if (!layer || layer.kind !== 'text') {
                throw new Error('非文本图层');
            }
            
            // 设置文本内容
            layer.textItem.contents = item.content;
            
            console.log(`[Filler] 文案已填充: ${item.layerName}`);
            
        }, { commandName: '填充文案' });
    }
    
    /**
     * 填充图片 (支持剪切蒙版)
     */
    private async fillImage(item: ImageFillItem): Promise<void> {
        await core.executeAsModal(async () => {
            const doc = app.activeDocument;
            const targetLayer = this.findLayerById(doc.layers, item.layerId);
            
            if (!targetLayer) {
                throw new Error('目标图层不存在');
            }
            
            // 获取目标边界
            const targetBounds = targetLayer.bounds;
            const targetWidth = targetBounds.right - targetBounds.left;
            const targetHeight = targetBounds.bottom - targetBounds.top;
            const targetCenterX = (targetBounds.left + targetBounds.right) / 2;
            const targetCenterY = (targetBounds.top + targetBounds.bottom) / 2;
            
            // 1. 置入图片
            const fileEntry = await fs.getEntryWithUrl('file:' + item.imagePath);
            const token = await fs.createSessionToken(fileEntry);
            
            const placeResult = await action.batchPlay([{
                _obj: 'placeEvent',
                null: { _path: token, _kind: 'local' },
                freeTransformCenterState: { _enum: 'quadCenterState', _value: 'QCSAverage' },
                offset: {
                    _obj: 'offset',
                    horizontal: { _unit: 'pixelsUnit', _value: targetCenterX },
                    vertical: { _unit: 'pixelsUnit', _value: targetCenterY }
                }
            }], { synchronousExecution: true });
            
            // 获取新置入的图层
            const placedLayer = doc.activeLayers[0];
            
            // 2. 缩放适配
            await this.scaleToFit(placedLayer, targetWidth, targetHeight, item.fillMode);
            
            // 3. 建立剪切蒙版 (如果需要)
            if (targetLayer.clipped || item.layerName.includes('图片')) {
                await this.moveLayerAbove(placedLayer, targetLayer);
                await this.createClippingMask(placedLayer);
            }
            
            // 4. 删除原占位图层
            await targetLayer.delete();
            
            console.log(`[Filler] 图片已填充: ${item.layerName}`);
            
        }, { commandName: '填充图片' });
    }
    
    /**
     * 缩放图层以适配目标尺寸
     */
    private async scaleToFit(
        layer: Layer, 
        targetWidth: number, 
        targetHeight: number,
        mode: 'cover' | 'contain' | 'smart'
    ): Promise<void> {
        const bounds = layer.bounds;
        const currentWidth = bounds.right - bounds.left;
        const currentHeight = bounds.bottom - bounds.top;
        
        let scale: number;
        
        if (mode === 'contain') {
            // 完整显示，可能有留白
            scale = Math.min(targetWidth / currentWidth, targetHeight / currentHeight);
        } else if (mode === 'cover') {
            // 填满区域，可能裁切
            scale = Math.max(targetWidth / currentWidth, targetHeight / currentHeight);
        } else {
            // smart: 根据主体位置智能决定
            scale = Math.max(targetWidth / currentWidth, targetHeight / currentHeight);
        }
        
        const scalePercent = scale * 100;
        
        await action.batchPlay([{
            _obj: 'transform',
            _target: [{ _ref: 'layer', _id: layer.id }],
            freeTransformCenterState: { _enum: 'quadCenterState', _value: 'QCSAverage' },
            width: { _unit: 'percentUnit', _value: scalePercent },
            height: { _unit: 'percentUnit', _value: scalePercent },
            interfaceIconFrameDimmed: { _enum: 'interpolationType', _value: 'bicubic' }
        }], { synchronousExecution: true });
    }
    
    /**
     * 移动图层到指定图层上方
     */
    private async moveLayerAbove(layer: Layer, targetLayer: Layer): Promise<void> {
        await action.batchPlay([{
            _obj: 'move',
            _target: [{ _ref: 'layer', _id: layer.id }],
            to: { _ref: 'layer', _id: targetLayer.id },
            adjustment: false
        }], { synchronousExecution: true });
    }
    
    /**
     * 建立剪切蒙版
     */
    private async createClippingMask(layer: Layer): Promise<void> {
        await action.batchPlay([{
            _obj: 'groupEvent',
            _target: [{ _ref: 'layer', _id: layer.id }]
        }], { synchronousExecution: true });
    }
    
    private findLayerById(layers: any[], id: number): any {
        for (const layer of layers) {
            if (layer.id === id) return layer;
            if (layer.layers) {
                const found = this.findLayerById(layer.layers, id);
                if (found) return found;
            }
        }
        return null;
    }
}
```

### 4.6 切片导出 (SliceExporter)

**文件**: `DesignEcho-UXP/src/tools/layout/slice-exporter.ts`

**导出流程**:
1. 按屏遍历
2. 隐藏其他屏
3. 裁切到当前屏边界
4. 导出 JPEG (JSX 方式)
5. 恢复可见性

**输出结构**:
```
项目目录/
└── 详情/
    ├── 01_核心卖点.jpg
    ├── 02_面料说明.jpg
    ├── 03_解决痛点.jpg
    └── ...
```

**核心实现**:

```typescript
// slice-exporter.ts (UXP 侧)
import { app, action, core } from 'photoshop';
import { saveAsJPEGViaJSX, ensureDirectoryViaJSX } from './export-folder-service';
import type { 
    ParsedScreen, SliceExportConfig, SliceExportResult, 
    ScreenExportResult 
} from './detail-page.types';

export class SliceExporter {
    
    /**
     * 导出所有屏为切片
     */
    async exportAll(
        screens: ParsedScreen[],
        config: SliceExportConfig
    ): Promise<SliceExportResult> {
        const startTime = Date.now();
        const results: ScreenExportResult[] = [];
        const errors: string[] = [];
        
        const doc = app.activeDocument;
        const originalState = await this.captureVisibilityState(doc);
        
        // 确保输出目录存在
        const outputDir = config.createSubfolder 
            ? `${config.outputDir}\\${config.subfolder}`
            : config.outputDir;
            
        await ensureDirectoryViaJSX(outputDir);
        
        try {
            for (let i = 0; i < screens.length; i++) {
                const screen = screens[i];
                
                try {
                    const result = await this.exportScreen(
                        screen, 
                        i, 
                        outputDir, 
                        config, 
                        doc
                    );
                    results.push(result);
                } catch (e: any) {
                    errors.push(`屏 ${i + 1} 导出失败: ${e.message}`);
                }
            }
        } finally {
            // 恢复原始可见性状态
            await this.restoreVisibilityState(doc, originalState);
        }
        
        return {
            success: errors.length === 0,
            screens: results,
            outputDir,
            totalScreens: screens.length,
            successCount: results.length,
            failedCount: errors.length,
            totalTime: Date.now() - startTime,
            errors: errors.length > 0 ? errors : undefined
        };
    }
    
    /**
     * 导出单个屏
     */
    private async exportScreen(
        screen: ParsedScreen,
        index: number,
        outputDir: string,
        config: SliceExportConfig,
        doc: Document
    ): Promise<ScreenExportResult> {
        
        await core.executeAsModal(async () => {
            // 1. 隐藏其他屏，只显示当前屏
            for (const layer of doc.layers) {
                if (layer.kind === 'group') {
                    layer.visible = (layer.id === screen.id);
                }
            }
            
            // 2. 裁切到当前屏边界
            const bounds = screen.bounds;
            await action.batchPlay([{
                _obj: 'crop',
                to: {
                    _obj: 'rectangle',
                    top: { _unit: 'pixelsUnit', _value: bounds.top },
                    left: { _unit: 'pixelsUnit', _value: bounds.left },
                    bottom: { _unit: 'pixelsUnit', _value: bounds.bottom },
                    right: { _unit: 'pixelsUnit', _value: bounds.right }
                },
                delete: true
            }], { synchronousExecution: true });
            
        }, { commandName: `裁切屏 ${index + 1}` });
        
        // 3. 生成文件名
        const fileName = this.generateFileName(screen, index, config.namingPattern);
        const filePath = `${outputDir}\\${fileName}.${config.format}`;
        
        // 4. 导出
        const saved = await saveAsJPEGViaJSX(filePath, config.quality);
        
        if (!saved) {
            throw new Error('导出失败');
        }
        
        // 5. 撤销裁切
        await core.executeAsModal(async () => {
            await action.batchPlay([{
                _obj: 'select',
                _target: [{ _ref: 'historyState', _property: 'previous' }]
            }], { synchronousExecution: true });
        }, { commandName: '撤销裁切' });
        
        console.log(`[SliceExporter] 导出成功: ${filePath}`);
        
        return {
            index,
            name: screen.name,
            type: screen.type,
            path: filePath,
            size: {
                width: bounds.width,
                height: bounds.height
            }
        };
    }
    
    /**
     * 生成文件名
     */
    private generateFileName(
        screen: ParsedScreen, 
        index: number, 
        pattern: string
    ): string {
        const paddedIndex = String(index + 1).padStart(2, '0');
        const safeName = screen.name.replace(/[\\/:*?"<>|]/g, '_');
        const typeShort = screen.type.split('_')[1] || screen.type;
        
        return pattern
            .replace('{index}', paddedIndex)
            .replace('{name}', safeName)
            .replace('{type}', typeShort);
    }
    
    /**
     * 捕获所有图层的可见性状态
     */
    private async captureVisibilityState(doc: Document): Promise<Map<number, boolean>> {
        const state = new Map<number, boolean>();
        
        const capture = (layers: any[]) => {
            for (const layer of layers) {
                state.set(layer.id, layer.visible);
                if (layer.layers) {
                    capture(layer.layers);
                }
            }
        };
        
        capture(doc.layers);
        return state;
    }
    
    /**
     * 恢复所有图层的可见性状态
     */
    private async restoreVisibilityState(
        doc: Document, 
        state: Map<number, boolean>
    ): Promise<void> {
        await core.executeAsModal(async () => {
            const restore = (layers: any[]) => {
                for (const layer of layers) {
                    const originalVisible = state.get(layer.id);
                    if (originalVisible !== undefined) {
                        layer.visible = originalVisible;
                    }
                    if (layer.layers) {
                        restore(layer.layers);
                    }
                }
            };
            
            restore(doc.layers);
        }, { commandName: '恢复可见性' });
    }
}
```

### 4.7 技能执行器 (DetailPageExecutor)

**文件**: `DesignEcho-Agent/src/renderer/services/skill-executors/detail-page.executor.ts`

**编排流程**:
```
1. 解析模板结构
2. 检测图层问题 → 自动修复
3. 生成填充方案
4. 遍历每屏:
   a. 执行填充
   b. 检测问题 → 修复
   c. (可选) VLM 验证
5. 批量导出切片
6. 返回结果报告
```

**核心实现**:

```typescript
// detail-page.executor.ts (Agent 侧)
import { ContentMatcher } from '@main/services/detail-page/content-matcher';
import { RAGService } from '@main/services/rag/rag-service';
import { EcommerceProjectService } from '@main/services/ecommerce-project-service';
import type { 
    DetailPageDesignParams, DetailPageDesignResult,
    ParsedScreen, FillPlan, LayerIssue, FixResult
} from '@shared/types/detail-page.types';

export class DetailPageExecutor {
    
    private contentMatcher: ContentMatcher;
    
    constructor(
        private uxpService: any,  // UXP 通信服务
        private ragService: RAGService,
        private projectService: EcommerceProjectService
    ) {
        this.contentMatcher = new ContentMatcher(ragService, projectService);
    }
    
    /**
     * 执行详情页设计
     */
    async execute(params: DetailPageDesignParams): Promise<DetailPageDesignResult> {
        const startTime = Date.now();
        const screenResults: DetailPageDesignResult['screens'] = [];
        let issuesFound = 0;
        let issuesFixed = 0;
        
        try {
            // ========== Phase 1: 模板解析 ==========
            console.log('[DetailPageExecutor] 开始解析模板...');
            const parseResult = await this.uxpService.invoke('detailPage.parse', {});
            
            if (!parseResult.success) {
                return this.errorResult('模板解析失败', startTime);
            }
            
            const screens: ParsedScreen[] = parseResult.screens;
            console.log(`[DetailPageExecutor] 解析完成: ${screens.length} 屏`);
            
            // ========== Phase 2: 初始问题检测 ==========
            console.log('[DetailPageExecutor] 检测图层问题...');
            const allIssues: LayerIssue[] = [];
            
            for (const screen of screens) {
                const issues = await this.uxpService.invoke(
                    'detailPage.detectIssues', 
                    { screenId: screen.id }
                );
                allIssues.push(...issues);
            }
            
            issuesFound = allIssues.length;
            console.log(`[DetailPageExecutor] 发现 ${issuesFound} 个问题`);
            
            // ========== Phase 3: 自动修复 ==========
            if (params.autoFix !== false && allIssues.some(i => i.autoFixable)) {
                console.log('[DetailPageExecutor] 执行自动修复...');
                const fixResults: FixResult[] = await this.uxpService.invoke(
                    'detailPage.fixAll',
                    { issues: allIssues.filter(i => i.autoFixable) }
                );
                issuesFixed = fixResults.filter(r => r.success).length;
                console.log(`[DetailPageExecutor] 修复完成: ${issuesFixed}/${allIssues.filter(i => i.autoFixable).length}`);
            }
            
            // ========== Phase 4: 生成填充方案 ==========
            console.log('[DetailPageExecutor] 生成填充方案...');
            const fillPlans = await this.contentMatcher.generateFillPlans(
                screens,
                params.projectPath
            );
            
            // ========== Phase 5: 逐屏填充 ==========
            console.log('[DetailPageExecutor] 开始填充...');
            
            for (let i = 0; i < screens.length; i++) {
                const screen = screens[i];
                const plan = fillPlans[i];
                
                console.log(`[DetailPageExecutor] 填充屏 ${i + 1}/${screens.length}: ${screen.name}`);
                
                // 执行填充
                const fillResult = await this.uxpService.invoke(
                    'detailPage.fill',
                    { plan }
                );
                
                // 填充后检测
                let postIssues: LayerIssue[] = [];
                if (fillResult.success) {
                    postIssues = await this.uxpService.invoke(
                        'detailPage.detectIssues',
                        { screenId: screen.id }
                    );
                    
                    // 修复填充产生的新问题
                    if (postIssues.length > 0 && params.autoFix !== false) {
                        await this.uxpService.invoke('detailPage.fixAll', {
                            issues: postIssues.filter(i => i.autoFixable)
                        });
                    }
                }
                
                screenResults.push({
                    index: i,
                    name: screen.name,
                    type: screen.type,
                    status: fillResult.success ? 
                        (postIssues.length > 0 ? 'warning' : 'success') : 'error',
                    issues: postIssues
                });
            }
            
            // ========== Phase 6: 导出切片 ==========
            console.log('[DetailPageExecutor] 导出切片...');
            const exportResult = await this.uxpService.invoke('detailPage.exportAll', {
                screens,
                config: {
                    outputDir: params.outputDir || `${params.projectPath}`,
                    format: params.exportFormat || 'jpeg',
                    quality: params.exportQuality || 10,
                    namingPattern: '{index}_{type}',
                    createSubfolder: true,
                    subfolder: '详情'
                }
            });
            
            // 更新导出路径
            for (const exportedScreen of exportResult.screens) {
                const screenResult = screenResults.find(s => s.index === exportedScreen.index);
                if (screenResult) {
                    screenResult.exportPath = exportedScreen.path;
                }
            }
            
            // ========== Phase 7: (可选) VLM 视觉验证 ==========
            if (params.visualValidation) {
                console.log('[DetailPageExecutor] VLM 视觉验证...');
                for (const result of screenResults) {
                    if (result.exportPath) {
                        const score = await this.validateWithVLM(result.exportPath);
                        result.score = score;
                        if (score < 0.6) {
                            result.status = 'warning';
                        }
                    }
                }
            }
            
            // ========== 返回结果 ==========
            const successCount = screenResults.filter(s => s.status !== 'error').length;
            const failedCount = screenResults.filter(s => s.status === 'error').length;
            
            return {
                success: failedCount === 0,
                message: `详情页设计完成: ${successCount}/${screens.length} 屏成功`,
                stats: {
                    screensProcessed: screens.length,
                    screensSuccess: successCount,
                    screensFailed: failedCount,
                    issuesFound,
                    issuesFixed,
                    totalTime: Date.now() - startTime
                },
                screens: screenResults,
                export: exportResult
            };
            
        } catch (e: any) {
            console.error('[DetailPageExecutor] 执行失败:', e);
            return this.errorResult(e.message, startTime);
        }
    }
    
    /**
     * VLM 视觉验证
     */
    private async validateWithVLM(imagePath: string): Promise<number> {
        try {
            // 调用视觉分析服务
            const analysis = await this.uxpService.invoke('vision.analyze', {
                imagePath,
                prompt: '评估这张详情页设计的视觉质量，包括排版、对比度、可读性，给出0-1的评分'
            });
            
            return analysis.score || 0.8;
        } catch {
            return 0.8;  // 默认分数
        }
    }
    
    private errorResult(message: string, startTime: number): DetailPageDesignResult {
        return {
            success: false,
            message,
            stats: {
                screensProcessed: 0,
                screensSuccess: 0,
                screensFailed: 0,
                issuesFound: 0,
                issuesFixed: 0,
                totalTime: Date.now() - startTime
            },
            screens: []
        };
    }
}
```
